\begin{frame}{Literature that shaped the panel (selected)}

    \item \textbf{Tian/Qiu 2023 (JoGH)} --- temperature variability and circulatory admissions $\rightarrow$ P15
    \item \textbf{Guo 2025} --- pollution amplifies temperature--hospitalisation associations $\rightarrow$ P11 staged
