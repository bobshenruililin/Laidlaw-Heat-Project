\begin{frame}{Literature that keeps us honest (selected)}
  \footnotesize
  \begin{itemize}
    \item \textbf{Goggins 2012/2013} --- cold-dominant patterns; subtype and sex matter
    \item \textbf{Guo 2024} --- official hot-night binary often weak; intensity / structure matter
    \item \textbf{Wang/Ren 2019} --- spells and combined day--night windows
    \item \textbf{Guo/Gasparrini 2017} --- percentile $\times$ duration grids; no single heatwave definition
    \item \textbf{Yang/Chong 2025} --- cold and influenza with stroke admissions
    \item \textbf{Guo 2025} --- pollution can amplify temperature--hospitalisation associations
  \end{itemize}
